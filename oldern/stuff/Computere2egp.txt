{
    if(chatClk(Target)){
    A = Target:lastSaid():explode(" ")
    if(A[1,string]:toNumber()){
        Num1 = A[1,string]:toNumber()
    }
    if(A[2,string]=="+"){
        Op = 1
    }elseif(A[2,string]=="-"){
        Op = 2
    }elseif(A[2,string]=="*"){
        Op = 3
    }elseif(A[2,string]=="/"){
        Op = 4
    }
    if(A[3,string]:toNumber()){
        Num2 = A[3,string]:toNumber()
    }
    if(Num1 & Num2 & Op){
        hideChat(1)
        if(Op==1){
            Range=(Num1+Num2)
            Range2 = Range:toString()
    CS:egpText(90,Range2,vec2(90,285))
    CS:egpColor(90,vec(20,20,20))
        }elseif(Op==2){
            Range=(Num1-Num2)
            Range2 = Range:toString()
    CS:egpText(90,Range2,vec2(90,285))
    CS:egpColor(90,vec(20,20,20))
        }elseif(Op==3){
            Range=(Num1*Num2)
            Range2 = Range:toString()
    CS:egpText(90,Range2,vec2(90,285))
    CS:egpColor(90,vec(20,20,20))

            Range=(Num1/Num2)
            Range2 = Range:toString()
    CS:egpText(90,Range2,vec2(90,285))
    CS:egpColor(90,vec(20,20,20))
        }
        Num1 = 0
        Num2 = 0
        Op = 0
    }
